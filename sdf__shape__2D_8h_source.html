97  coord_2d grad_sdf(const coord_2d& pt) const {
98  double x = pt[0] - center[0];
99  double y = pt[1] - center[1];
100  double r = sqrt(x*x + y*y);
101  coord_2d g;
102  if(r < 1.0e-6) {
103  g[0] = g[1] = 0.0;
104  }
105  else {
106  g[0] = x/r;
107  g[1] = y/r;
108  }
109  return g;
110  }
